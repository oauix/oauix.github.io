Socket与ServerSocket

Socket封装了TCP通讯协议,使用它可以基于TCP协议与远端计算机通讯.

	import java.net.Socket;
	class Client{//客户端
		private Socket socket;
		Client(){
			socket=new Socket("localhost",8088);//初始化socket，并创建服务器连接信息
		}
		public static void main(String[] agrs){
			Client cli=new Client();
		}
	}

	class Server{//服务器端
		private Socket ServerSocket serversocket;
		Server(){
			serversocket=new ServerSocket(8088);//初始化,并向系统申请8088端口号
		}
		void start(){
			Socket socket=serversocket.aceep();//监听客户端，若有客户端，建立连接
		}
		public static void main(String[] agrs){
			Server sv=new Server();
		}
	}

.getInputStream() 获取输入流

	InputStream in=socket.getInputStream();//获取输入
	InputStreamReader isr=new InputStreamReader(in,"utf-8");//转换流
	BufferedReader br=new BufferedReader(isr);//字符缓存流
	br.readLine();//将读取第一行打印出来

.getOutputStream() 创建输出流


	OutputStream out=socket.getOutputStream();//创建输出流
	OutputStreamWriter osw=new OutputStreamWriter(out,"utf-8");//转换流
	PrintWriter pw=new PrintWriter(osw,true);//字符缓存流，并开启自动刷新
	pw.println("需要输出第一行内容");
