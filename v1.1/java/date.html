Date

*Date库类(从java1.1已不用了)

getTime()与setTime();前者将date类型值转换为long，后者相反

	import java.util.Date;
	/**
	*使用Date输出当前系统时间，并告知3天后这一刻的时间
	*
	*/

   public class test{
		public static void main(String[] agrs){
			Date date=new Date();//获取一个时间节点，当前系统时间
			System.out.println(date);
			long time = date.getTime();//转换为自1970.1.1到截取时间的毫秒数
			System.out.println(time);
			time+=24*60*60*1000*3;
			date.setTime(time);//再转换为日期格式
			System.out.println(date);
		}
   }

SimpleDateFormat[格式类，java.text.SimpleDateFormat]

自定义格式：object.format();[日期格式>自定义]object.parse();[相反]

	
	import java.util.Date;
	import java.text.SimpleDateFormat;
	/**
	*将当前系统时间转换以"yyyy-MM-dd HH:mm:ss"格式输出
	*
	*/
	public class tests{
		public static void main(String[] agrs){
			Date date = new Date();
			SimpleDateFormat sd = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
			String str = sd.format(date);//将日期格式转换为指定格式
			Date sb= sd.parse(str);//再将指定格式反转换为日期格式
			System.out.println(str+","+sb);
		}
	}
