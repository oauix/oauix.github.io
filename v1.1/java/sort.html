	//假设数组
	int[] arr={12，20,123,20}
	/**
	*每次内循环一次，找出最大的值放在后面
	*/
	//对比次数
	for(int i=0;i<arr.lenght-1;i++){
		//到后面时，前面一判定最后几个是最大值，再比就无意义，故-i;
		for(int j=0;j<arr.lenght-1-i;j++){
				//假如前值大于后面值，就换位置。
				if(arr[j]>arr[j+1]){
					//追尾法
					int che=arr[j];//以此类推，那个最大值就到最后面；
					arr[j]=arr[j+1];
					arr[j+1]=che;
				}
		}
	}
