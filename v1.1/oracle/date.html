日期操作：

	1.last_day(date); 获取这个月最后一天

	2.add_months(date,i); 获取日期date加上i个月的日期值

	3.month_between(date1,date2);获取date1与date2之间间隔多少月

	4.NEXT_day(date,n)
	5.select NEXT_day(to_date('2017-05-17','YYYY-MM-DD'),1) FROM DUAL;--从明天开始一周内的周日是哪天

	6.extract (date from datetime) 提取年月日，
		select id,name from dual where extract(year from sysdate)=2017;
		select extract(hour from timestamp '2008-10-10 10:10:10') dual;
